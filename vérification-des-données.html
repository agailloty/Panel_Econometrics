replacements <- list( 
  pct = c("%", "pct"),
  USD = c("\\$", "d"),
  coma = c(",", "_"),
  open_bracket = c("\\(", ""),
  closed_bracket = c("\\)", ""),
  white_space = c(" ", "_")
  )

clean_names <- function(x, patternReplacement = replacements) {
  out <- tolower(x)
  for (repl in patternReplacement) {
    out <- gsub(pattern = repl[1], replacement = repl[2], x = out)
  }
  out
}
colnames(ok) <- clean_names(colnames(ok))
